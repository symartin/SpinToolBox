00371 QString QTechSpinBox::textFromValue(double value) const{
00372    // string with at least sufficient numer of digits, separated in mantissa and exponent
00373    
00374    QString str = locale().toString(value, 'e', postdigitsInt + predigitsInt + 2);
00375    int exponent = str.section(locale().exponential(), 1, 1).toInt();
00376    str = str.section(locale().exponential(), 0, 0); //mantissa (could be negative) with decimal sign
00377    int signSpace_1(str.at(0) == locale().negativeSign() ? 1 + 1 : 0 + 1); // current position of dec'point
00378    Q_ASSERT(str.contains(locale().decimalPoint()));
00379    str = str.remove(locale().decimalPoint());                           // remove delimiter (normally '.' or ',')
00380    // determine new position of decimal sign, and position correctly
00381    int shiftDelimiter;
00382    switch (predigitsInt){
00383       case 0:
00384          shiftDelimiter = -1;  //remove standard single digit in front of delimiter
00385          break;
00386       case 1:
00387          shiftDelimiter = 0;
00388          break;
00389       default:
00390          if (exponent >= 0) shiftDelimiter = exponent % predigitsInt;
00391          else {
00392             shiftDelimiter = abs(exponent) % predigitsInt;
00393             if (shiftDelimiter > 0) shiftDelimiter = predigitsInt - shiftDelimiter;}
00394          break;}
00395    exponent -= shiftDelimiter;
00396    int str_size = shiftDelimiter + signSpace_1 + postdigitsInt;
00397    Q_ASSERT(str.size() > (str_size + 1));    // at least two digits more than needed, to allow rounding
00398    unsigned short int cutDigit(locale().toInt((str.at(str_size))));
00399    unsigned short int followingDigit(locale().toInt(str.at(str_size+1)));
00400    str.resize(str_size);
00401    qlonglong mantissaInt(locale().toLongLong(str));
00402    switch (roundingruleEnum){
00403    // should check that ALL following digits = 0 if cutDigit=5; for double: check only one following digit
00404       case Cut:
00405          break;
00406       case BankersRule:      // to even number
00407          if (cutDigit > 5 ||
00408             (cutDigit == 5 && followingDigit > 0) ||
00409             (cutDigit == 5 && followingDigit == 0 && (abs(mantissaInt) % 2) != 0)){
00410          mantissaInt >= 0 ? ++mantissaInt : --mantissaInt;}
00411          break;
00412       case Arithmetic:       // as symmetric, but -3.50 gives -3
00413          if (cutDigit > 5 ||
00414             (cutDigit == 5 &&  followingDigit > 0) ||
00415             (cutDigit == 5 && mantissaInt >= 0 && followingDigit == 0))
00416          mantissaInt >= 0 ? ++mantissaInt : --mantissaInt;
00417          break;
00418       case Symmetric:       // if cut digit is 5 or above; negative and positives are rounding oppositely
00419          if (cutDigit >= 5) mantissaInt >= 0 ? ++mantissaInt : --mantissaInt;
00420          break;}
00421    str = locale().toString(mantissaInt).trimmed();
00422    str = str.remove(locale().groupSeparator());
00423    //prepend '0' (locale-aware) to fill up to required number of digits (postdigits + 1)
00424    for (int i(str.size()); i < postdigitsInt + 1; ++i) str.prepend(locale().toString(0));
00425    if (postdigitsInt > 0) {
00426        /*
00427       Q_ASSERT_X(str.size() > shiftDelimiter + signSpace_1, str.toAscii(),
00428               QString("%1 + %2").arg(shiftDelimiter).arg(signSpace_1).toAscii());
00429       */
00430       str = str.insert(shiftDelimiter + signSpace_1, locale().decimalPoint());}
00431    if (!baseunitString.isEmpty() && predigitsInt == 3 && unitmodifiersBool == true &&
00432       exponent >= lowestPrefix && exponent <= highestPrefix){  // unit prefix
00433       Q_ASSERT_X(abs(exponent) % 3 == 0, "% 3 == 0", QString(exponent).toAscii());
00434       str = str + QChar(QLatin1Char(' ')) + QString(unitPrefix[(exponent - lowestPrefix) / 3] + baseunitString).trimmed();}
00435    else {
00436       str += QString(QLatin1String("%1%2")).arg(locale().exponential()).arg(locale().toString(exponent));
00437       if (!baseunitString.isEmpty()) str += QLatin1String(" ") + baseunitString;}
00438    return str;}

00448 double QTechSpinBox::valueFromText(const QString &text) const {
00449    double mantissa;
00450    int exponent(0);
00451    int firstNonNumeric(0);
00452    while (firstNonNumeric < text.size() && isNumeric(text.at(firstNonNumeric))) ++firstNonNumeric;
00453    bool ok;
00454    mantissa = locale().toDouble(text.left(firstNonNumeric), &ok);
00455    Q_ASSERT(ok);
00456    if (firstNonNumeric == text.size()) return mantissa;
00457    QString expString = text.mid(firstNonNumeric).trimmed(); //head of expString: e, unit, modifier
00458    if (expString.size() > 1 && expString.at(0) == locale().exponential() && isNumeric(expString.at(1))) {
00459       expString = expString.mid(1);    // cut 'e' or locale corresponding
00460       firstNonNumeric = 0;
00461       while (firstNonNumeric < expString.size() && isNumeric(expString.at(firstNonNumeric)))
00462                                                                                        ++firstNonNumeric;
00463       exponent = locale().toInt(expString.left(firstNonNumeric), &ok);
00464       Q_ASSERT_X(ok, "toInt", expString.toAscii());
00465       expString = expString.mid(firstNonNumeric).trimmed();}
00466    if (expString.size() > 0 && !baseunitString.startsWith(expString)){  // not only baseunit
00467       int unitNumber(0);
00468       while (unitPrefix[unitNumber] != expString[0]) ++unitNumber;
00469       Q_ASSERT_X(unitNumber <= (highestPrefix - lowestPrefix), "unit prefix not found: ", QString(expString[0]).toAscii());
00470       exponent += 3 * unitNumber + lowestPrefix;}
00471    mantissa = mantissa * pow((double)10, (double)exponent);
00472    return mantissa;}

00474 void QTechSpinBox::setRegExpValidator(){
00475    // does lineEdit take over ownership of QRegExpValidator? is nowhere deleted!
00476    QString regExp = QString(QLatin1String("^%1?")).arg(QRegExp::escape(locale().negativeSign()));
00477    if (predigitsInt == 0) ;      // no digits prior to decimal sign
00478    else if (predigitsInt == 3 && !baseunitString.isEmpty()) regExp += QString(QLatin1String("\\d{1,3}"));
00479    else regExp += QString(QLatin1String("\\d{1,%1}")).arg(predigitsInt);
00480    // add optional decimalPoint and digits (digits mandatory, if decimalPoint present)
00481    if (postdigitsInt > 0) regExp += QString(QLatin1String("(%1\\d{1,%2})?")).arg(QRegExp::escape(locale().decimalPoint()))
00482                                                              .arg(postdigitsInt);
00483    // add optional exponent (incl negativeSign) and optional trailing space
00484    regExp += QString(QLatin1String("(e%1?\\d{1,3})? ?")).arg(QRegExp::escape(locale().negativeSign()));
00485    // add optional unit prefix, optional unit
00486    QString allUnitPrefixes(QLatin1String(""));
00487    for (int i(0); i <= (highestPrefix - lowestPrefix) / 3; ++i)
00488    if (!unitPrefix[i].isSpace()) allUnitPrefixes += unitPrefix[i];
00489    regExp += QString(QLatin1String(QLatin1String("[%1]?"))).arg(QRegExp::escape(allUnitPrefixes));
00490    if (!baseunitString.isEmpty()){   // allow any part from start of baseunit
00491       for (int i(0); i < baseunitString.size(); ++i)
00492                   regExp += QString(QLatin1String(QLatin1String("(%1"))).arg(QRegExp::escape(QString(baseunitString[i])));
00493       for (int i(0); i < baseunitString.size(); ++i) regExp += QLatin1String(")?");}
00494    regExp += QString(QLatin1String("$"));
00495    this->lineEdit()->setValidator(new QRegExpValidator(QRegExp(regExp), 0));
00496    updateDisplay();}
